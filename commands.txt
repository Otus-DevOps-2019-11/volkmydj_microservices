###Create machine
docker-machine create --driver google --google-machine-image https://www.googleapis.com/compute/v1/projects/ubuntu-os-cloud/global/images/family/ubuntu-1604-lts --google-machine-type n1-standard-1 --google-zone europe-west1-b docker-host
gcloud compute firewall-rules create reddit-app --allow tcp:9292 --target-tags=docker-machine --description="Allow PUMA connections" --direction=INGRESS

docker tag reddit:latest volkmydj/otus-reddit:1.0
docker push volkmydj/otus-reddit:1.0
docker run --name reddit -d -p 9292:9292 volkmydj/otus-reddit:1.0
docker run --name reddit --rm -it volkmydj/otus-reddit:1.0 bash

### Посмотреть контейнер
docker inspect volkmydj/otus-reddit:1.0
docker inspect volkmydj/otus-reddit:1.0 -f '{{.ContainerConfig.Cmd}}'

### В контейнере
docker exec -it reddit bash
